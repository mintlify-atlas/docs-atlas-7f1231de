## Overview

This guide provides practical examples for using the Beaver Builder REST API endpoints. All examples use real endpoint patterns from the source code.

## Available Endpoints

The Beaver Builder REST API provides three main endpoints:

- **GET** `/wp-json/fl-builder/v1/{page_id}` - Get complete page content (HTML, CSS, and JS)
- **GET** `/wp-json/fl-builder/v1/css/{page_id}` - Get only CSS for a page
- **GET** `/wp-json/fl-builder/v1/js/{page_id}` - Get only JavaScript for a page

<Note>
  All endpoints accept a numeric `page_id` parameter representing the WordPress post/page ID.
</Note>

## Getting Complete Page Content

The main endpoint returns a complete representation of a Beaver Builder page, including HTML content, CSS styles, and JavaScript.

### Endpoint Details

**URL Pattern:** `/wp-json/fl-builder/v1/{page_id}`

**Response Object:**
```json
{
  "id": 123,
  "html": "<div class='fl-builder-content'>...</div>",
  "css": ".fl-node-abc123{...}",
  "js": "jQuery(document).ready(function(){...})"
}
```

## Getting Page CSS Only

Use this endpoint when you only need the CSS styles for a page. The CSS is returned as a minified string with whitespace removed.

### Endpoint Details

**URL Pattern:** `/wp-json/fl-builder/v1/css/{page_id}`

**Response:** Minified CSS string

<Tip>
  The CSS endpoint returns a plain string (not JSON) with all whitespace, newlines, and tabs removed for optimal size.
</Tip>

### Examples

<CodeGroup>

```bash cURL
# Get CSS for page 123
curl https://example.com/wp-json/fl-builder/v1/css/123

# With authentication
curl -u username:app-password \
  https://example.com/wp-json/fl-builder/v1/css/123

# Save to file

  https://example.com/wp-json/fl-builder/v1/css/123 \
  -o page-123.css
```

```javascript JavaScript

    'Authorization': `Basic ${btoa('username:app-password')}`

  .then(css => {
    const style = document.createElement('style');
    style.textContent = css;
    document.head.appendChild(style);

  });
```

## Real-World Use Cases

### Headless CMS Integration

Use Beaver Builder as a headless CMS for your frontend application:

```javascript
class BeaverBuilderClient {
  constructor(baseUrl, username, password) {
    this.baseUrl = baseUrl;
    this.credentials = btoa(`${username}:${password}`);
  }

  async getPage(pageId) {
    const response = await fetch(
      `${this.baseUrl}/wp-json/fl-builder/v1/${pageId}`,
      {
        headers: {
          'Authorization': `Basic ${this.credentials}`
        }
      }
    );
    return response.json();
  }

  async renderPage(pageId, containerId) {
    const data = await this.getPage(pageId);
    
    // Inject HTML
    document.getElementById(containerId).innerHTML = data.html;
    
    // Add CSS
    const style = document.createElement('style');
    style.id = `bb-css-${pageId}`;
    style.textContent = data.css;
    document.head.appendChild(style);
    
    // Execute JS
    const script = document.createElement('script');
    script.id = `bb-js-${pageId}`;
    script.textContent = data.js;
    document.body.appendChild(script);
  }
}

// Usage
const bb = new BeaverBuilderClient(
  'https://example.com',
  'api-user',
  'xxxx xxxx xxxx xxxx'
);

bb.renderPage(123, 'app-container');
```

### Static Site Generation

Export Beaver Builder pages to static HTML files:

```python
import requests
from requests.auth import HTTPBasicAuth
import os

class BeaverBuilderExporter:
    def __init__(self, base_url, username, password):
        self.base_url = base_url
        self.auth = HTTPBasicAuth(username, password)
    
    def export_page(self, page_id, output_dir):
        """Export a page to static HTML file"""
        url = f"{self.base_url}/wp-json/fl-builder/v1/{page_id}"
        response = requests.get(url, auth=self.auth)
        
        if response.status_code != 200:
            raise Exception(f"Failed to fetch page {page_id}")
        
        data = response.json()
        
        # Create HTML file with inline CSS and JS
        html_content = f"""
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>Page {data['id']}</title>
    <style>
        {data['css']}
    </style>
</head>
<body>
    {data['html']}
    <script>
        {data['js']}
    </script>
</body>
</html>
        """
        
        os.makedirs(output_dir, exist_ok=True)
        output_file = os.path.join(output_dir, f"page-{page_id}.html")
        
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(html_content)
        
        return output_file
    
    def export_multiple(self, page_ids, output_dir):
        """Export multiple pages"""
        results = []
        for page_id in page_ids:
            try:
                file_path = self.export_page(page_id, output_dir)
                results.append({'id': page_id, 'success': True, 'file': file_path})
                print(f"Exported page {page_id} -> {file_path}")
            except Exception as e:
                results.append({'id': page_id, 'success': False, 'error': str(e)})
                print(f"Failed to export page {page_id}: {e}")
        return results

# Usage
exporter = BeaverBuilderExporter(
    'https://example.com',
    'api-user',
    'xxxx xxxx xxxx xxxx'
)

# Export multiple pages
page_ids = [123, 456, 789]
results = exporter.export_multiple(page_ids, './static-pages')
```

### Content Backup System

Create automated backups of Beaver Builder pages:

```javascript
const axios = require('axios');
const fs = require('fs').promises;
const path = require('path');

class BeaverBuilderBackup {
  constructor(baseUrl, username, password) {
    this.client = axios.create({
      baseURL: baseUrl,
      auth: { username, password }
    });
  }

  async backupPage(pageId, backupDir) {
    try {
      const response = await this.client.get(
        `/wp-json/fl-builder/v1/${pageId}`
      );
      
      const { id, html, css, js } = response.data;
      const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
      const pageDir = path.join(backupDir, `page-${id}`);
      
      await fs.mkdir(pageDir, { recursive: true });
      
      // Save each component separately
      await Promise.all([
        fs.writeFile(
          path.join(pageDir, `${timestamp}.html`),
          html
        ),
        fs.writeFile(
          path.join(pageDir, `${timestamp}.css`),
          css
        ),
        fs.writeFile(
          path.join(pageDir, `${timestamp}.js`),
          js
        ),
        fs.writeFile(
          path.join(pageDir, `${timestamp}.json`),
          JSON.stringify(response.data, null, 2)
        )
      ]);
      
      return { success: true, pageId: id, timestamp };
    } catch (error) {
      return {
        success: false,
        pageId,
        error: error.message
      };
    }
  }

  async backupMultiple(pageIds, backupDir) {
    const results = await Promise.all(
      pageIds.map(id => this.backupPage(id, backupDir))
    );
    return results;
  }
}

// Usage
const backup = new BeaverBuilderBackup(
  'https://example.com',
  'api-user',
  'xxxx xxxx xxxx xxxx'
);

backup.backupMultiple([123, 456, 789], './backups')
  .then(results => {
    results.forEach(r => {
      if (r.success) {
        console.log(`Backed up page ${r.pageId} at ${r.timestamp}`);
      } else {
        console.error(`Failed to backup page ${r.pageId}: ${r.error}`);
      }
    });
  });
```

## Error Handling

Always implement proper error handling in your API requests:

<CodeGroup>

```javascript JavaScript
async function fetchPage(pageId) {
  try {
    const response = await fetch(
      `https://example.com/wp-json/fl-builder/v1/${pageId}`,
      {
        headers: {
          'Authorization': `Basic ${credentials}`
        }
      }
    );

    if (!response.ok) {
      if (response.status === 404) {
        throw new Error(`Page ${pageId} not found`);
      } else if (response.status === 401) {
        throw new Error('Authentication failed');
      } else if (response.status === 403) {
        throw new Error('Access forbidden');
      }
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }

    return await response.json();
  } catch (error) {
    console.error('Error fetching page:', error);
    throw error;
  }
}
```

```python Python
import requests
from requests.auth import HTTPBasicAuth
from requests.exceptions import RequestException

def fetch_page(page_id, base_url, username, password):
    url = f"{base_url}/wp-json/fl-builder/v1/{page_id}"
    
    try:
        response = requests.get(
            url,
            auth=HTTPBasicAuth(username, password),
            timeout=30
        )
        response.raise_for_status()
        return response.json()
        
    except requests.exceptions.HTTPError as e:
        if e.response.status_code == 404:
            print(f"Page {page_id} not found")
        elif e.response.status_code == 401:
            print("Authentication failed")
        elif e.response.status_code == 403:
            print("Access forbidden")
        else:
            print(f"HTTP error: {e}")
        raise
        
    except RequestException as e:
        print(f"Request failed: {e}")
        raise
```

## Performance Tips

<Tip>
  For better performance when fetching multiple pages, use parallel requests with proper rate limiting.
</Tip>

```javascript
// Batch fetch with concurrency control
async function fetchPages(pageIds, concurrency = 3) {
  const results = [];
  
  for (let i = 0; i < pageIds.length; i += concurrency) {
    const batch = pageIds.slice(i, i + concurrency);
    const promises = batch.map(id => fetchPage(id));
    const batchResults = await Promise.allSettled(promises);
    results.push(...batchResults);
  }
  
  return results;
}
```
